        {

            scanf("%d", &arr[i]);

        int max_indx = 0, min_indx = 0;
    
        for (int i = 0; i < n; i++)
        {
            if (arr[i] > max)

                max = arr[i];

            }

                min = arr[i];

    int main()
    {
        int n;
        scanf("%d", &n);
    
        for (int i = 1; i <= n; i++)
        {
            for (int j = 1; j <= n; j++)
            {
                if (j == i || j == n - i + 1)
                {
                    printf("x");
                }
                else
                {
                    printf(" ");
                }
            }
            printf("\n");
        }
     return 0;
       }
