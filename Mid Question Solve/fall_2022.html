           scanf("%d", &n);
       
           int max_even = 0;
           int max_evn_index = 0;
       
           for (int i = 0; i < n; i++)
           {
               scanf("%d", &marks[i]);
       
                   if (marks[i] % 2 == 0 && (marks[i] > max_even))
                   {
       
                       max_even = marks[i];
                       max_evn_index = i;
                   }

       }

       {
       
           int n;
           scanf("%d", &n);
       
           for (int i = 1; i <= n; i++)
           {
               for (int j = 1; j <= n; j++)
               {
                   if ((i <= n / 2 )&& (i == j || j == n - i + 1))
                   {
                       printf("* ");
                   }
                   else if ((i>n/2) && (j == n / 2 + 1))
                   {
                       printf("* ");
                   }
                   else
                   {
                       printf("  ");
                   }
                   
               }
               printf("\n");
           }
               return 0;
